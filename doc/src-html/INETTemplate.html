010public interface INETTemplate
011{
012    /**
013     * Accept a full file path and name and determine if it exists on disk.
014     * If it does, return a true. If it does not exist, return a false.
015     * @param strFileName -
016     * @return
017     */
018    public Boolean fileExist(String strFileName);
019
020    /**
021     * * Accept a full file path and name as a parameter and
022     * check to see if it exists using the fileExists method.
023     * If it does exist, read the file from disk and return the contents as a string.
024     * If the file does not exist, return a null.
025     * @param strFileName - a full file path and name
026     * @return - Contents from the file as a string.
027     * @throws java.lang.Exception
028     */
029    java.lang.String getFromFile(java.lang.String strFileName)
030            throws java.lang.Exception;
031
032    /**
033     * Accept an HTML web page as a string and
034     * extract the data that is between the PRE tags using the getRegEx function.
035     * If nothing is found, return a null.
036     * @param strPage - a normal HTML page as a string
037     * @return - Extracted Contents - only the data between two PRE tags or null.
038     */
039    public String getPREData(java.lang.String strPage);
040
041    /**
042     * Extract a small string from a larger one using a Regular Expression and
043     * the pattern specified as a parameter.
044     * If nothing is found, return a null.
045     * @param strInput - large string
046     * @param strPattern - formatted Regular Expression
047     * @return - a small string that is the result of the Reg Ex search
048     */
049    public String getRegEx(java.lang.String strInput, java.lang.String strPattern);
050
051
052    /**
053     *  Accept a string URL, create a string using the StringBuilder class and
054     * return it as one long string. If the page does not exist, return a null.
055     * @param strURL - URL as a string
056     * @return -  the entire web page as a string
057     * @throws java.lang.Exception
058     */
059    java.lang.String getURLRaw(java.lang.String strURL)
060            throws java.lang.Exception;
061
062    /**
063     * Accept a string and convert the whole string into lower case then
064     * upper case only the first letter of each word.
065     *
066     *
067     * @param strInput - string to be converted
068     * @return - string with every word's first letter capitalized
069     */
070    public String properCase(String strInput);
071
072
073    /**
074     * Accept a string and save it to a local file
075     * @param strFileName - a full file path and name.
076     * @param strContent - string to be saved to disk.
077     * @throws java.lang.Exception
078     */
079    void saveToFile(java.lang.String strFileName,
080                    java.lang.String strContent)
081            throws java.lang.Exception;
082
083
084}
