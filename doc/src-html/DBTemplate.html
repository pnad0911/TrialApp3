011public interface DBTemplate
012{
013    /**
014     * The addRecord method checks to see if a specific record already exists
015     * in the database and if it doesn't, it adds it.
016     * If the record already exists, it just skips over all of the code
017     * @param strTable - the table
018     * @param strKeyName - the key name
019     * @param strKeyContents - the key contents of the key field
020     * @return - true or fail;
021     */
022    public Boolean addRecord(String strTable,
023                             String strKeyName,
024                             String strKeyContents);
025
026    /**
027     * The close method basically tells the database to
028     * write everything to disk and clear the cache
029     */
030    public void close();
031
032    /**
033     * delete table
034     * @param strTable - table name
035     * @return
036     */
037    public Boolean deleteAll(String strDataName,String strTable);
038
039    /**
040     *
041     * @param strFieldName
042     * @return
043     */
044    public String getField(String strFieldName);
045
046    /**
047     *First you run the query, then you loop through the database checking each time to see
048     * if there are any moreRecords()
049     * @return - returns TRUE if there are more records and FALSE if there are not
050     */
051    public Boolean moreRecords();
052
053    /**
054     * the openConnection method accepts the Data Source Name of the database
055     * then looks for a Java Derby connector
056     * @param strDataSourceName - the Data Source Name of the database
057     */
058    public Boolean openConnection(String strDataSourceName);
059
060    /**
061     * the query method asks the database for information
062     * but does not make any changes to the database.
063     * @param strSQL - SQL
064     */
065    public Boolean query(String strSQL);
066
067    /**
068     * To update a record with multiple fields, you simply call the method once for each field
069     * @param strTable - the table
070     * @param strKeyName - the key name
071     * @param strKeyContents - the key contents of the key field
072     * @param strFieldName - the field name
073     * @param strFieldContents - the contents of the key field
074     * @return - true or false
075     */
076    public Boolean setField(String strTable,
077                            String strKeyName,
078                            String strKeyContents,
079                            String strFieldName,
080                            String strFieldContents);
081
082    /**
083     * In some cases you need to create a database from scratch.  To do that we need to have a function
084     * that allows changes to the database
085     * @param strSQL - SQL
086     * @return - true or false
087     */
088    public Boolean execute(String strSQL);
089}
