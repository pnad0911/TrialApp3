099    /**
100     * Create stringSQL
101     * @param strTable - table name
102     * @param strFileName - file name
103     * @return - String SQL
104     * @throws Exception
105     */
106    public static String stringSQL(String strTable, String strFileName) throws Exception {
107        INET net = new INET();
108        String strPage;
109        //check to see if the file exists on disk
110        Boolean boTest = net.fileExist(strFileName);
111        if (boTest == false) {
112            if (strFileName.equals("data/Grades.txt")) {
113                strPage = net.getURLRaw("http://faculty.sdmiramar.edu/jcouture/2014sp/cisc190/webct/manual/schema-grades.txt");
114                //strTest = net.getPREData(strPage);
115                net.saveToFile(strFileName, strPage);
116            } else if (strFileName.equals("data/Roster.txt")) {
117                //download the Winds Aloft data from the NWS and save it
118                strPage = net.getURLRaw("http://faculty.sdmiramar.edu/jcouture/2014sp/cisc190/webct/manual/schema-roster.txt");
119                //strTest = net.getPREData(strPage);
120                net.saveToFile(strFileName, strPage);
121            }
122        }
123        String str = "";
124        File myFile = new File(strFileName);
125        if (myFile.exists()) {
126            int intCount = 0;
127            Scanner inputFile = new Scanner(myFile);
128            StringBuilder strTGD = new StringBuilder("CREATE TABLE " + strTable + " (");
129            while (inputFile.hasNext()) {
130                String strLine;
131                strLine = inputFile.nextLine();
132                if (intCount > 9) {
133                    String[] tokens = strLine.split("\t");
134                    strTGD.append(tokens[0] + " ");
135                    if (strTable.equals(strTableGrades)) {
136                        aryGrades.add(tokens[0]);
137                    } else {
138                        aryRoster.add(tokens[0]);
139                    }
140                    String strL=tokens[1].substring(3, tokens[1].length());
141                    StringBuilder strL1=new StringBuilder(strL);
142                    if(strL.indexOf("l")>-1)
143                    {
144                        strL1.deleteCharAt(strL.indexOf("l"));
145                    }
146                    strTGD.append(strL1.toString().toUpperCase());
147                    if (tokens[2].trim().equals("NO") & intCount == 10) {
148                        strTGD.append(" NOT NULL");
149                        if (tokens[3].trim().equals("PRI")) {
150                            strTGD.append(" PRIMARY KEY, ");
151                        } else {
152                            strTGD.append(", ");
153                        }
154                    } else {
155                        strTGD.append(", ");
156                    }
157                }
158                intCount++;
159            }
160            strTGD.replace(strTGD.length() - 2, strTGD.length() - 1, ")");
161            str = strTGD.toString();
162        }
163
164        return str;
165    }

248    /**
249     * getQuizTotal(strStuID) that will return a "float" representing the total points of the student's quizzes
250     * @param strStuID - student ID
251     * @return - Total points of student's quiz
252     */
253    public static float getQuizTotal(String strStuID) {
254        DBUpdt db = new DBUpdt();
255        db.openConnection(strData);
256        //find stuid
257        db.query("SELECT * FROM " + strTableGrades + " WHERE stuid = '" + strStuID + "'");
258        int intCou;
259        float fl;
260        float tt = 0000;
261        while (db.moreRecords()) {
262            for (intCou = 21; intCou < 36; intCou++) {
263                String strS = db.getField(aryGrades.get(intCou));
264                //Divide that score by 100 and multiply it by 2
265                fl = (Float.valueOf(strS) / 100) * 2;
266                tt = tt + fl;
267            }
268        }
269        DecimalFormat format = new DecimalFormat("###.##");
270        float fltt=Float.valueOf(format.format(tt));
271        return fltt;
272    }

274    /**
275     * getAsgnTotal(strStuID) that will return a "float"  representing the total points of the student's assignments
276     * @param strStuID - student ID
277     * @return -  total points of the student's assignments
278     */
279    public static float getAsgnTotal(String strStuID) {
280        int intCou;
281        int intCou1 = 0;
282        float fl;
283        float tt = 0000;
284        //Initialize an array of point values for each assignment using the JCouture.net course overview page as your guide
285        String[] ary = new String[15];
286        String strValueAsgn = "22233344455679";
287        int intC;
288        for (intC = 0; intC < strValueAsgn.length(); intC++) {
289            ary[intC] = String.valueOf(strValueAsgn.charAt(intC));
290        }
291        ary[14] = "11";
292        DBUpdt db = new DBUpdt();
293        db.openConnection(strData);
294        //Look up the student grades in the GRADES table
295        db.query("SELECT * FROM " + strTableGrades + " WHERE stuid = '" + strStuID + "'");
296        while (db.moreRecords()) {
297            for (intCou = 6; intCou < 21; intCou++) {
298                //Calculate the number points the student earned by taking the assignment score and dividing it by 100 then
299                // multiply it by the number of points for that assignment
300                String strS = db.getField(aryGrades.get(intCou));
301                fl = (Float.valueOf(ary[intCou1]) / 100) * (Float.valueOf(strS));
302                tt = tt + fl;
303                intCou1++;
304            }
305        }
306        DecimalFormat format = new DecimalFormat("###.##");
307        float fltt=Float.valueOf(format.format(tt));
308        return fltt;
309    }

311    /**
312     * getLetterGrade(fltTotal) that accepts the total point value as a float and
313     * returns a single letter string representing the ordinal grade
314     * @param fltTotal - total points
315     * @return - letter grade
316     */
317    public static String getLetterGrade(float fltTotal) {
318        String strLetterGrade = "";
319        if (fltTotal < 60) {
320            strLetterGrade = "F";
321        } else if (fltTotal >= 60 & fltTotal < 70) {
322            strLetterGrade = "D";
323        } else if (fltTotal >= 70 & fltTotal < 80) {
324            strLetterGrade = "C";
325        } else if (fltTotal >= 80 & fltTotal < 90) {
326            strLetterGrade = "B";
327        } else if (fltTotal >= 90) {
328            strLetterGrade = "A";
329        }
330        return strLetterGrade;
331    }

334     * Create Report
335     */
336    public static void CreateReport() {
337        DBUpdt db = new DBUpdt();
338        db.openConnection(strData);
339        db.query("SELECT roster.*, grades.* FROM " + strTableRoster + " INNER JOIN grades ON roster.stuid=grades.stuid ORDER BY lastname,firstname");
340        //write a report to a file called SEMESTER.TXT
341        File file = new File("data/SEMESTER.TXT");
342        PrintWriter output = null;
343        try {
344            // write a file to disk using the filename
345            output = new PrintWriter(file);
346        } catch (Exception egg) {
347            egg.printStackTrace();
348        }
349        while (db.moreRecords()) {
350            DecimalFormat format = new DecimalFormat("#0.00");
351            float flAsgn = getAsgnTotal(db.getField("stuid"));
352            float flQuiz = getQuizTotal(db.getField("stuid"));
353            output.println(db.getField("lastname").trim() + ";" + db.getField("firstname").trim()
354                    + ";" + db.getField("stuid").trim() + ";" + db.getField("email").trim() + ";" +
355                    String.valueOf(format.format(flQuiz)) + ";" + String.valueOf(format.format(flAsgn)) + ";" +
356                    String.valueOf(format.format((flAsgn + flQuiz))) + ";" + getLetterGrade(flAsgn + flQuiz));
357        }
358        db.close();
359        output.close();
360    }

438        public void actionPerformed(ActionEvent e) {
439            String selection13;
440            selection13 = (String) comboBox1.getSelectedItem();

442            if (blnChange(selection13) == true) {
443                DBUpdt db = new DBUpdt();
444                db.openConnection(strData);
445                db.query("SELECT * FROM " + strTableRoster + " WHERE lastname = '" + tokens[0] + "' AND firstname ='" + tokens[1] + "'");
446                while (db.moreRecords()) {
447                    String strID13 = db.getField("stuid");
448                    lblStuID.setText(strID13);
449                    DecimalFormat format = new DecimalFormat("#0.00");
450                    float flAsgn = getAsgnTotal(strID13);
451                    float flQuiz = getQuizTotal(strID13);
