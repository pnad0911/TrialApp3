014public class dialogBox extends JDialog
015{
016    private JPanel contentPane;
017    private JTextArea textArea1;
018    private JButton OKbutton;
019    private JButton buttonOK;
020    public String strFileName1;
021
022    public dialogBox()
023    {
024        setContentPane(contentPane);
025        setModal(true);
026        getRootPane().setDefaultButton(buttonOK);
027        OKbutton.addActionListener(new test1());
028    }
029
030    public static void main(String[] args)
031    {
032        dialogBox dialog = new dialogBox();
033        dialog.pack();
034        dialog.setVisible(true);
035        System.exit(0);
036    }
037
038    /**
039     * Public inner class that handle the event when
040     * the user clicks a button
041     */
042    public class test1 implements ActionListener
043    {
044        public void actionPerformed(ActionEvent e)
045        {
046            strFileName1=textArea1.getText();
047            onCancel();
048            setDefaultCloseOperation(DO_NOTHING_ON_CLOSE);
049            addWindowListener(new WindowAdapter() {
050                public void windowClosing(WindowEvent e) {
051                    onCancel();
052                }
053            });
054
055// call onCancel() on ESCAPE
056            contentPane.registerKeyboardAction(new ActionListener() {
057                public void actionPerformed(ActionEvent e) {
058                    onCancel();
059                }
060            }, KeyStroke.getKeyStroke(KeyEvent.VK_ESCAPE, 0), JComponent.WHEN_ANCESTOR_OF_FOCUSED_COMPONENT);
061        }
062    }
063    private void onCancel()
064    {
065// add your code here if necessary
066        dispose();
067    }
068}
